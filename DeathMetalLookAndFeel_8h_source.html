316 void drawGroupComponentOutline(juce::Graphics &g, int width, int height, const juce::String &text,
317 const juce::Justification &position, juce::GroupComponent &group) override
318 {
319 const float textH = 15.0f;
320 const float indent = 3.0f;
321 const float textEdgeGap = 4.0f;
322 auto cs = 5.0f;
323
324 Path p;
325 auto x = indent;
326 auto y = deathMetalFont.getAscent() - 3.0f;
327 auto w = jmax(0.0f, (float)width - x * 2.0f);
328 auto h = jmax(0.0f, (float)height - y - indent);
329 cs = jmin(cs, w * 0.5f, h * 0.5f);
330 auto cs2 = 2.0f * cs;
331
332 auto textW = text.isEmpty() ? 0
333 : jlimit(0.0f, jmax(0.0f, w - cs2 - textEdgeGap * 2),
334 (float)deathMetalFont.getStringWidth(text) + textEdgeGap * 2.0f);
335 auto textX = cs + textEdgeGap;
336
337 if (position.testFlags(Justification::horizontallyCentred))
338 textX = cs + (w - cs2 - textW) * 0.5f;
339 else if (position.testFlags(Justification::right))
340 textX = w - cs - textW - textEdgeGap;
341
342 p.startNewSubPath(x + textX + textW, y);
343 p.lineTo(x + w - cs, y);
344
345 p.addArc(x + w - cs2, y, cs2, cs2, 0, MathConstants<float>::halfPi);
346 p.lineTo(x + w, y + h - cs);
347
348 p.addArc(x + w - cs2, y + h - cs2, cs2, cs2, MathConstants<float>::halfPi, MathConstants<float>::pi);
349 p.lineTo(x + cs, y + h);
350
351 p.addArc(x, y + h - cs2, cs2, cs2, MathConstants<float>::pi, MathConstants<float>::pi * 1.5f);
352 p.lineTo(x, y + cs);
353
354 p.addArc(x, y, cs2, cs2, MathConstants<float>::pi * 1.5f, MathConstants<float>::twoPi);
355 p.lineTo(x + textX, y);
356
357 auto alpha = group.isEnabled() ? 1.0f : 0.5f;
358
359 g.setColour(group.findColour(GroupComponent::outlineColourId).withMultipliedAlpha(alpha));
360
361 g.strokePath(p, PathStrokeType(2.0f));
362
363 g.setColour(group.findColour(GroupComponent::textColourId).withMultipliedAlpha(alpha));
364 g.setFont(deathMetalFont);
365 g.drawText(text, roundToInt(x + textX), 0, roundToInt(textW), roundToInt(textH), Justification::centred, true);
366 }
